/** * 依照檔案「定價總表」與「產品成本」定義的配置 */ const DEFAULT_CONFIG = { pricing: { L: 16800, H: 18800, DISCOUNT: 10800 }, commissionRate: 0.05, // 檔案：佣金 5% unitSoftwareCost: 2352, // 檔案：軟體服務單位成本 staffCapacity: 500, // 檔案：每人服務 500 家 monthlySalaryPerStaff: 35000, // 檔案：基本薪資 35,000 laborBurdenMultiplier: 1.8, // 假設：勞健保與獎金係數 (使 35k 換算後符合檔案勞務成本) fixedCosts: { hosting: 360000 / 12, // 檔案：AWS 360,000/年 bandwidth: 12000 / 12, // 檔案：網路頻寬 12,000/年 devAmortization: 1500000 / 12, // 檔案：軟體開發費用分攤 } }; /** * 完全依照檔案算式的計算引擎 */ function calculateGCFinancials(monthlySales: { L: number; H: number; DISCOUNT: number }, config = DEFAULT_CONFIG) { const { L, H, DISCOUNT } = monthlySales; const totalQty = L + H + DISCOUNT; // 1. 營收計算 (Revenue) const revenue = (L * config.pricing.L) + (H * config.pricing.H) + (DISCOUNT * config.pricing.DISCOUNT); // 2. 變動成本 (Variable Costs) const commission = revenue * config.commissionRate; const softwareServiceCost = totalQty * config.unitSoftwareCost; // 3. 人力成本 (Labor Cost) - 檔案邏輯：累積客戶數 / 服務人力 const requiredStaff = Math.ceil(totalQty / config.staffCapacity); // 檔案 2025 年 4 人對應 3,024,000，換算每月每人約 63,000 (含勞健保/獎金) const monthlyLaborCost = requiredStaff * 63000; // 4. 固定成本 (Fixed Costs) const totalFixedCost = config.fixedCosts.hosting + config.fixedCosts.bandwidth + config.fixedCosts.devAmortization; // 5. 毛利與淨利 const grossProfit = revenue - commission - softwareServiceCost - monthlyLaborCost; const netProfit = grossProfit - totalFixedCost; return { revenue, costs: { commission, softwareServiceCost, labor: monthlyLaborCost, fixed: totalFixedCost }, profit: { grossProfit, netProfit, margin: revenue > 0 ? grossProfit / revenue : 0 }, metrics: { requiredStaff, BEP_Qty: Math.ceil(totalFixedCost / (config.pricing.L - (config.pricing.L * config.commissionRate) - config.unitSoftwareCost)) } }; }
